## Quick Start

```rust
use praisonai::Agent;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let agent = Agent::new()
        .name("Assistant")
        .build()?;
    
    agent.chat("Hello!").await?;
    Ok(())
}
```

## Quick Start

<Steps>
<Step title="Analyze Video">
```rust
use praisonai::{Agent, tool};

#[tool]
async fn analyze_video(path: String) -> String {
    // Video analysis
    video::analyze(&path).await
}

let agent = Agent::new()
    .name("Video Analyst")
    .tool(analyze_video)
    .build()?;

agent.chat("Describe this video: demo.mp4").await?;
```

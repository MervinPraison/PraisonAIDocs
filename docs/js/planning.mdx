## Plans

### Create a Plan

```typescript
import { Plan, PlanStep } from 'praisonai';

const plan = new Plan({ name: 'Project Setup' });

plan.addStep(new PlanStep({ description: 'Initialize repository' }));
plan.addStep(new PlanStep({ description: 'Install dependencies' }));
plan.addStep(new PlanStep({ description: 'Configure environment' }));

console.log(`Plan: ${plan.name}`);
console.log(`Steps: ${plan.steps.length}`);
```

### Execute Steps

```typescript
import { Plan, PlanStep } from 'praisonai';

const plan = new Plan({ name: 'Build Process' });
plan.addStep(new PlanStep({ description: 'Compile code' }));
plan.addStep(new PlanStep({ description: 'Run tests' }));
plan.addStep(new PlanStep({ description: 'Deploy' }));

plan.start();

for (const step of plan.steps) {
  step.start();
  // Execute step logic...
  step.complete();
}

plan.complete();
```

### Track Progress

```typescript
import { Plan, PlanStep } from 'praisonai';

const plan = new Plan({ name: 'Migration' });
plan.addStep(new PlanStep({ description: 'Step 1' }));
plan.addStep(new PlanStep({ description: 'Step 2' }));
plan.addStep(new PlanStep({ description: 'Step 3' }));

plan.steps[0].complete();
plan.steps[1].complete();

const progress = plan.getProgress();
console.log(`Progress: ${progress.completed}/${progress.total} (${progress.percentage}%)`);
```

## Todo Lists

### Create a Todo List

```typescript
import { TodoList, TodoItem } from 'praisonai';

const todos = new TodoList('Daily Tasks');

todos.add(new TodoItem({ content: 'Review PRs', priority: 'high' }));
todos.add(new TodoItem({ content: 'Write tests', priority: 'medium' }));
todos.add(new TodoItem({ content: 'Update docs', priority: 'low' }));

console.log(`Tasks: ${todos.items.length}`);
```

### Manage Items

```typescript
import { TodoList, TodoItem } from 'praisonai';

const todos = new TodoList();

const task = new TodoItem({ content: 'Important task', priority: 'high' });
todos.add(task);

// Start working on it
task.start();
console.log(task.status); // 'in_progress'

// Complete it
task.complete();
console.log(task.status); // 'completed'
```

### Filter Items

```typescript
import { TodoList, TodoItem } from 'praisonai';

const todos = new TodoList();
todos.add(new TodoItem({ content: 'Task 1', priority: 'high' }));
todos.add(new TodoItem({ content: 'Task 2', priority: 'low' }));
todos.add(new TodoItem({ content: 'Task 3', priority: 'high' }));

// Get high priority items
const highPriority = todos.getByPriority('high');
console.log(`High priority: ${highPriority.length}`);

// Get pending items
const pending = todos.getPending();

// Get completed items
const completed = todos.getCompleted();
```

## Plan Storage

```typescript
import { PlanStorage, Plan, TodoList } from 'praisonai';

const storage = new PlanStorage();

// Save a plan
const plan = new Plan({ name: 'My Plan' });
storage.savePlan(plan);

// Retrieve a plan
const retrieved = storage.getPlan(plan.id);

// List all plans
const allPlans = storage.listPlans();

// Save a todo list
const todos = new TodoList('My Todos');
storage.saveTodoList(todos);

// List all todo lists
const allTodos = storage.listTodoLists();
```

## Data Structures

### PlanStep

```typescript
interface PlanStep {
  id: string;
  description: string;
  status: 'pending' | 'in_progress' | 'completed' | 'failed' | 'cancelled';
  order: number;
  dependencies: string[];
  metadata: Record<string, any>;
  startedAt?: number;
  completedAt?: number;
}
```

### TodoItem

```typescript
interface TodoItem {
  id: string;
  content: string;
  priority: 'low' | 'medium' | 'high';
  status: 'pending' | 'in_progress' | 'completed';
  dueDate?: Date;
  metadata: Record<string, any>;
  createdAt: number;
  completedAt?: number;
}
```

## Factory Functions

```typescript
import { createPlan, createTodoList, createPlanStorage } from 'praisonai';

const plan = createPlan({ name: 'New Plan' });
const todos = createTodoList('New List');

```

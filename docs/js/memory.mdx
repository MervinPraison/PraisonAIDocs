## Installation

```bash
npm install praisonai
```

## Basic Usage

```typescript
import { Memory } from 'praisonai';

const memory = new Memory();

await memory.add('Hello, how are you?', 'user');
await memory.add('I am doing well, thank you!', 'assistant');

console.log(`Memory size: ${memory.size}`);
```

## Search Memory

```typescript
import { Memory } from 'praisonai';

const memory = new Memory();

await memory.add('TypeScript is great for type safety', 'user');
await memory.add('Python is excellent for data science', 'user');
await memory.add('JavaScript runs in browsers', 'user');

const results = await memory.search('TypeScript');
results.forEach(r => {
  console.log(`Score: ${r.score}, Content: ${r.entry.content}`);
});
```

## Get Recent Entries

```typescript
import { Memory } from 'praisonai';

const memory = new Memory();

await memory.add('Message 1', 'user');
await memory.add('Message 2', 'assistant');
await memory.add('Message 3', 'user');

const recent = memory.getRecent(2);
console.log(`Last 2 messages: ${recent.length}`);
```

## Build Context

```typescript
import { Memory } from 'praisonai';

const memory = new Memory();

await memory.add('What is AI?', 'user');
await memory.add('AI is artificial intelligence', 'assistant');

const context = memory.buildContext();
console.log(context);
// user: What is AI?
// assistant: AI is artificial intelligence
```

## With Embedding Provider

```typescript
import { Memory } from 'praisonai';

const embeddingProvider = {
  embed: async (text: string) => {
    // Your embedding logic
    return [0.1, 0.2, 0.3];
  },
  embedBatch: async (texts: string[]) => {
    return texts.map(() => [0.1, 0.2, 0.3]);
  }
};

const memory = new Memory({
  maxEntries: 1000,
  embeddingProvider
});
```

## Export/Import

```typescript
import { Memory } from 'praisonai';

const memory = new Memory();
await memory.add('Test', 'user');

// Export
const data = memory.toJSON();

// Import
const newMemory = new Memory();
newMemory.fromJSON(data);
```

## Factory Function

```typescript
import { createMemory } from 'praisonai';

const memory = createMemory({
  maxEntries: 500,
  maxTokens: 50000
});
```

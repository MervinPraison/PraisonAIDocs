PraisonAI TypeScript provides an evaluation framework for testing AI agents.

## Installation

```bash
npm install praisonai
```

## Accuracy Evaluation

```typescript
import { accuracyEval } from 'praisonai';

const result = await accuracyEval({
  input: 'What is 2+2?',
  expectedOutput: 'The answer is 4',
  actualOutput: 'The answer is 4',
  threshold: 0.8
});

console.log('Passed:', result.passed);
console.log('Score:', result.score);
```

## Performance Evaluation

```typescript
import { performanceEval } from 'praisonai';

const result = await performanceEval({
  func: async () => {
    return await agent.chat('Hello');
  },
  iterations: 10,
  warmupRuns: 2
});

console.log('Avg Time:', result.avgTime, 'ms');
console.log('Min Time:', result.minTime, 'ms');
console.log('Max Time:', result.maxTime, 'ms');
console.log('P95 Time:', result.p95Time, 'ms');
```

## Reliability Evaluation

```typescript
import { reliabilityEval } from 'praisonai';

const result = await reliabilityEval({
  expectedToolCalls: ['calculator', 'search'],
  actualToolCalls: ['calculator', 'search', 'extra']
});

console.log('Passed:', result.passed);
console.log('Score:', result.score);
console.log('Missing:', result.details?.missing);
console.log('Extra:', result.details?.extra);
```

## Eval Suite

```typescript
import { EvalSuite } from 'praisonai';

const suite = new EvalSuite();

// Run multiple evaluations
await suite.runAccuracy('accuracy-test', {
  input: 'test',
  expectedOutput: 'hello',
  actualOutput: 'hello'
});

await suite.runPerformance('perf-test', {
  func: async () => await agent.chat('Hi'),
  iterations: 5
});

await suite.runReliability('reliability-test', {
  expectedToolCalls: ['tool1'],
  actualToolCalls: ['tool1']
});

// Get summary
const summary = suite.getSummary();
console.log('Total:', summary.total);
console.log('Passed:', summary.passed);
console.log('Failed:', summary.failed);
console.log('Avg Score:', summary.avgScore);

// Print detailed results
suite.printSummary();
```

## Custom Thresholds

```typescript
const result = await accuracyEval({
  input: 'test',
  expectedOutput: 'expected result',
  actualOutput: 'actual result',
  threshold: 0.5 // Lower threshold for partial matches
});
```

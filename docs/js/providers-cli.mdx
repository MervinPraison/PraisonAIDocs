```bash
# Set API keys
export OPENAI_API_KEY=your-key
export ANTHROPIC_API_KEY=your-key
export GOOGLE_API_KEY=your-key
```

```bash
# Run with OpenAI (default)
npx ts-node examples/simple/basic.ts
```

```bash
# Run with specific provider
PRAISONAI_MODEL=anthropic/claude-sonnet-4-20250514 npx ts-node examples/simple/basic.ts
```

```bash
# Run with Google
PRAISONAI_MODEL=google/gemini-2.0-flash npx ts-node examples/simple/basic.ts
```

```bash
# Check available providers
npx ts-node -e "const {getAvailableProviders} = require('praisonai'); console.log(getAvailableProviders())"
```

```bash
# Test provider connection
npx ts-node tests/development/verify-providers.ts
```

---
title: "Prompt Expander CLI"
description: "CLI commands for prompt expansion"
---

```bash
# Expand a prompt with default settings
npx ts-node -e "
import { PromptExpanderAgent } from 'praisonai';
const agent = new PromptExpanderAgent();
agent.expand('Write code').then(r => console.log(r.expanded));
"

# Expand with detail strategy
npx ts-node -e "
import { PromptExpanderAgent } from 'praisonai';
const agent = new PromptExpanderAgent();
agent.expand('Create function', 'detail').then(r => console.log(r.expanded));
"

# Expand with context strategy
npx ts-node -e "
import { PromptExpanderAgent } from 'praisonai';
const agent = new PromptExpanderAgent();
agent.expand('Explain this', 'context').then(r => console.log(r.expanded));
"

# Expand with examples strategy
npx ts-node -e "
import { PromptExpanderAgent } from 'praisonai';
const agent = new PromptExpanderAgent();
agent.expand('Show me', 'examples').then(r => console.log(r.expanded));
"

# Detect strategy for a prompt
npx ts-node -e "
import { PromptExpanderAgent } from 'praisonai';
const agent = new PromptExpanderAgent();
console.log(agent.detectStrategy('Write code'));
"
```

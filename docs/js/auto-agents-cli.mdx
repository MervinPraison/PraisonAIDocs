---
title: "AutoAgents CLI"
description: "CLI commands for automatic agent generation"
---

```bash
# Generate agents for a task
npx ts-node -e "
import { AutoAgents } from 'praisonai';
const auto = new AutoAgents();
auto.generate('Build a web scraper').then(team => {
  console.log('Agents:', team.agents.length);
  console.log('Pattern:', team.pattern);
});
"

# Recommend pattern
npx ts-node -e "
import { AutoAgents } from 'praisonai';
const auto = new AutoAgents();
console.log(auto.recommendPattern('Run in parallel'));
"

# Analyze complexity
npx ts-node -e "
import { AutoAgents } from 'praisonai';
const auto = new AutoAgents();
console.log(auto.analyzeComplexity('Simple task'));
console.log(auto.analyzeComplexity('Complex multi-step pipeline'));
"

# Single agent mode
npx ts-node -e "
import { AutoAgents } from 'praisonai';
const auto = new AutoAgents({ singleAgent: true });
auto.generate('Write code').then(team => {
  console.log('Agents:', team.agents.length);
});
"

# With specific pattern
npx ts-node -e "
import { AutoAgents } from 'praisonai';
const auto = new AutoAgents({ pattern: 'parallel' });
auto.generate('Process data').then(team => {
  console.log('Pattern:', team.pattern);
});
"
```
